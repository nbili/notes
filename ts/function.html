# 函数重载

function sum(...rest: number[]): number
function sum(...rest: string[]): string
function sum(...rest: any[]) {
  let x = rest[0]
  if (typeof x === 'number') {
    return rest.reduce((pre, cur) => pre + cur)
  }
  if (typeof x === 'string') {
    return rest.join(' ')
  }
}
console.log(sum(1, 3, 4)) // 8
console.log(sum('1', '2', '3')) // '1 2 3'
